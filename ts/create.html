# typescript测试项目创建

// ts安装
yarn add typescript -g
// ts初始化
tsc --init
// eslint安装
yarn add eslint

yarn add typescript typescript-eslint-parser eslint-plugin-typescript

//创建eslint配置文件
./node_modules/.bin/eslint --init

// 编辑.eslintrc.js 增加配置parser: "typescript-eslint-parser"
